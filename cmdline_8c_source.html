58 int rb_push(cmd_line_desc_ptr_t cmd_line_desc, char c) {
59  int next = cmd_line_desc->rx_buf_head + 1;
60  if (next >= cmd_line_desc->rx_buf_max_len) {
61  next = 0;
62  }
63 
64  if (next == cmd_line_desc->rx_buf_tail) {
65  return -1;
66  }
67 
68  cmd_line_desc->rx_buf[cmd_line_desc->rx_buf_head] = c;
69  cmd_line_desc->rx_buf_head = next;

71  return 0;

89 int cmd_line_printf_parse(cmd_line_desc_ptr_t cmd_line_desc, const char *format, ...) {
90  va_list args;
91  va_start (args, format);
92  int ret_val = vsnprintf(cmd_line_desc->tx_buf, cmd_line_desc->tx_buf_len, format, args);
93  if (ret_val < 0) {

98  }
99  va_end (args);

171 int cmd_line_process(cmd_line_desc_ptr_t cmd_line_desc) {
172  int retVal = CMD_LINE_NO_CMD;
173  int8_t c = cmd_line_getchar(cmd_line_desc);
174  if (c == EOF) {
175  return retVal;
176  }
177 
178  if (isprint(c)) {
179  if (cmd_line_desc->line_buf_current < cmd_line_desc->line_buf_len) {
180  if (cmd_line_desc->echo_enabled) {
181  cmd_line_printf_tk(cmd_line_desc, "%c", c, 1000);
182  }
183  cmd_line_desc->line_buf[cmd_line_desc->line_buf_current++] = c;
184  }
185  }
186  else if (c == '\r') {
187  if (cmd_line_desc->echo_enabled) {
188  cmd_line_printf_tk(cmd_line_desc, "%c", c, 1000);
189  }
190  }
191  else if (c == '\n') {
192  if (cmd_line_desc->echo_enabled) {
193  cmd_line_printf_tk(cmd_line_desc, "%c", c, 1000);
194  }
195  cmd_line_desc->line_buf[cmd_line_desc->line_buf_current] = '\0';
196  int i;
197  int argc = 0;
198  cmd_line_desc->arg_starts[argc] = &cmd_line_desc->line_buf[0];
199  for (i = 0; i < cmd_line_desc->line_buf_current; i++) {
200  if (cmd_line_desc->line_buf[i] == ' ') {
201  cmd_line_desc->line_buf[i] = '\0';
202  argc++;
203  cmd_line_desc->arg_starts[argc] = &cmd_line_desc->line_buf[i+1];
204  }
205  }
206  cmd_line_desc->line_buf_current = 0;
207  const cmd_desc_t* cmd_ptr = cmd_line_find_command_by_name(cmd_line_desc->cmd_line_root_list, cmd_line_desc->arg_starts[0], 0);
208  if (cmd_ptr != NULL) {
209  retVal = cmd_ptr->cmd_fcn(argc, cmd_line_desc->arg_starts, cmd_line_desc);
210  }
211  else {
212  retVal = CMD_LINE_ERR_CMD_NOT_FOUND;
213  }
214  }
215  return retVal;
216 }
